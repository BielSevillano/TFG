int mcd(int a, int b)
{
    int auxiliar;
    if (b > a) {
        auxiliar = a;
        a = b;
        b = auxiliar;
    }

    if (b == 0) {
        return a;
    }
    if (a % b == 0) {
        return b;
    } else {
        return mcd(b, a % b);
    }
}
