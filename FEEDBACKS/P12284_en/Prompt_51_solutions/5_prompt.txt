'''

'''

n = read(int) #num of words
while n is not None:
    k = read(int) #num of most frequent words

    cnt = 0
    word_dict = {}
    while cnt < n:
        "makes a dictionary of how many times a word appears"
        word = read(str)
        if word not in word_dict.keys():
            word_dict[word] = 1
        else:
            word_dict[word] += 1
        cnt += 1


    num_dict = {}
    cnt_2 = 1
    while cnt_2 <= max(word_dict.values()):
        "makes a dictionary of which words appear a certain number of times"
        for value in word_dict:
            if word_dict[value] == cnt_2:
                num_dict[cnt_2] = num_dict.setdefault(cnt_2, []) + [value]
        cnt_2 += 1


    new_list = []
    for i in sorted(num_dict.keys(), reverse = True):
        "makes a sorted list of the words in the way we wanted, from most to less frequent ones and in alphabetical order in case of a tie"
        new_list.extend(sorted(num_dict[i]))


    cnt_k = 0
    while cnt_k < k:
        "prints the number of most frequent words we specified with 'k'"
        print(new_list[cnt_k])
        cnt_k += 1

    print(10*'-')

def comptar_paraules(n):
    dic = {}
    for i in range(n):
        pal = read(str)
        if pal in dic:
            dic[pal] += 1
        else:
            dic[pal] = 1
    return dic

def generar_llista_ordenada(dic):
    l = []
    for pal,freq in dic.items():
        l.append((pal,freq))
    l.sort(key = itemgetter(0))
    #print(l)
    l.sort(key = itemgetter(1), reverse = True)
    return l

def escriure_k_primers(l,k):

def set_dictionary(n):
    words={}
    final_dict={}
    for k in range(n):
        word=read(str)
        if word not in words:
            words[word]=1
        else:
            words[word]+=1   #create the dictionary
            
    
    for key in sorted(words, reverse=True):
        if words[key] not in final_dict:
            final_dict[words[key]]=[key]
        else:
            aux = final_dict[words[key]]
            
            aux.append(key)
            final_dict[words[key]]=aux

    return final_dict       #reversed dictionary (to sort by values the main dictionary)
