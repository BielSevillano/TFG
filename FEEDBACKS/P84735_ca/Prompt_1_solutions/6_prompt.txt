                # Programa que calcule potencias 
# L'entrada consisteix en 2 numeros parells a i b
# Cada parell a, b , a es elevat al valor de b

def potencia(a, b):
    if b == 0:
        return 1
    elif b == 1:
        return a
    else:
        return a * potencia(a, b - 1)


def main():
    while True:
        a, b = map(int, input().split())
        if a == 0 and b == 0:
            break
        print(potencia(a, b))
