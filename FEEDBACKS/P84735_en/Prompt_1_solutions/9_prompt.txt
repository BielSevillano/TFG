case number starting at 1, following the format of the example.

int mcm (int max, int min){
    if (min == 1 or max == 1) return max*min;
    if (max%min == 0) return max;
    for(int i =2;i*i<max;++i) if(i*min>=max) return i*min;
    return max*min;
}
