    def fusio(v1: list[float], v2: list[float]) -> list[float]:
        i1 = 0
        i2 = 0
        llista_definitiva = []
        while i1 < len(v1) and i2 < len(v2):
            if v1[i1] < v2[i2]:
                llista_definitiva.append(v1[i1])
                i1 += 1
            else:
                llista_definitiva.append(v2[i2])
                i2 += 1

        llista_definitiva.extend(v2[i2:])
