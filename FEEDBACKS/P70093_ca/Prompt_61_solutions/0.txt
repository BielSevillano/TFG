def fusio(v1: list[float], v2: list[float]) -> list[float]:
    """
    Merges two sorted lists (v1 and v2) into a single sorted list,
    preserving duplicate elements.

    Precondition: v1 and v2 are sorted in ascending order.
    Example: fusio([2, 3, 5], [3, 5, 5, 7]) -> [2, 3, 3, 5, 5, 5, 7]
    """
    i1 = 0
    i2 = 0
    llista_definitiva = []

    while i1 < len(v1) and i2 < len(v2):
        if v1[i1] < v2[i2]:
            llista_definitiva.append(v1[i1])
            i1 += 1
        else: # Handles v1[i1] >= v2[i2], appending v2[i2] first on equality
            llista_definitiva.append(v2[i2])
            i2 += 1

    # Append any remaining elements from v1
    llista_definitiva.extend(v1[i1:])
    # Append any remaining elements from v2
    llista_definitiva.extend(v2[i2:])

    return llista_definitiva
