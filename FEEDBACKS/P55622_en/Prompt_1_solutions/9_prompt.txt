                This is the program:

                import sys

def numdigits(n):
    digits = 0
    for i in n:
        digits = len(n)
    return digits

n = sys.stdin.readline().strip()
x = numdigits(n)
print("The number of digits of " + str(n) + " is " + str(x) + ".")
