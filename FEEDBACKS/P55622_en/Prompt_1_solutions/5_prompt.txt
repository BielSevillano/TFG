                from easyinput import read

def digits(n):
    digits=1;
    n=n//10;
    while n>0:
        digits= digits +1
        n=n//10
    return digits

def main():
    n=read(int)
    print("The number of digits of ", n, " is ", digits(n), ".", sep="")

main()
