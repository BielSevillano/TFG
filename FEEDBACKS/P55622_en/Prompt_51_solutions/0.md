    ```python
    from easyinput import read

    n = read(int)
    count = 0

    if n == 0:
        count = 1
    else:
        temp_n = n  # Use a temporary variable to preserve the original number for output

            temp_n //= 10
            count += 1

    print("The number of digits of", n, "is", count, ".", sep="")
    ```
